Psych Score Analysis & Pandemic Event Impact

Assessment of scale reliability (Cronbach's Alpha) and linear regression modeling of survey constructs.

DATA CLEANING & PREPARING

library(tidyverse)
library(psych)

# Load data (Encrypted path)
data_path <- "encrypted_path/Psych Score_data.csv"
dat <- read_csv(data_path)

# Data is already pre-processed with generic names:
# Psych Score_score, gender, university_type, age_num, etc.

# Create Psych Score Categories
dat <- dat %>%
  mutate(
    Psych Score_cat = case_when(
      is.na(Psych Score_score) ~ NA_character_,
      Psych Score_score < 2.5  ~ "Low",
      Psych Score_score < 3.5  ~ "Moderate",
      TRUE                ~ "High"
    ) |> factor(levels = c("Low", "Moderate", "High"))
  )

INFERENTIALS 

# 1. Reliability Analysis (Cronbach's Alpha)
alpha_items <- dat %>% select(item_1, item_2, item_3, item_4) %>% drop_na()
psych::alpha(alpha_items)$total

# 2. Group Comparisons
# T-test: Psych Score by Gender
t.test(Psych Score_score ~ gender, data = dat)

# ANOVA: Psych Score by University Type
summary(aov(Psych Score_score ~ university_type, data = dat))

# 3. Multiple Linear Regression
model <- lm(Psych Score_score ~ gender + university_type + age_num + study_hours, data = dat)
summary(model)

CHI-SQUARE ANALYSIS

library(effectsize)

# Chi-square Helper Function
run_chi_square <- function(var_name) {
  tbl <- table(dat[[var_name]], dat$Psych Score_cat)
  
  # Use Fisher's Exact Test if small sample size
  if (any(chisq.test(tbl)$expected < 5)) {
    test <- fisher.test(tbl)
    method <- "Fisher's Exact"
  } else {
    test <- chisq.test(tbl)
    method <- "Chi-square"
  }
  
  list(
    Variable = var_name,
    Method = method,
    P_Value = test$p.value,
    Cramers_V = cramers_v(tbl)$Cramers_v
  )
}

# Run for Pandemic Event-related variables
Pandemic Event_vars <- c("Pandemic Event_positive", "family_positive", "isolation_feeling")
results <- map_dfr(Pandemic Event_vars, run_chi_square)
print(results)

SENSITIVITY ANALYSIS

# Multicollinearity Check (VIF)
library(car)
vif(model)

# P-value Adjustment (Benjamini-Hochberg)
results$p_adj <- p.adjust(results$P_Value, method = "BH")
print(results)
