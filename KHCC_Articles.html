Data Cleaning

# ============================================
# DATA CLEANING (ENCRYPTED)
# ============================================

library(readxl)
library(dplyr)
library(lubridate)
library(stringr)

# Load data (Encrypted path)
data_path <- "encrypted_path/hct_cohort.xlsx"
raw_df <- read_excel(data_path)

# Create analysis dataset with masked variable names
df <- raw_df %>%
  mutate(
    birth_date = as.Date(birth_date),
    index_date = as.Date(index_date),
    relapse_date = as.Date(relapse_date),
    death_date = as.Date(death_date),
    last_follow_up = as.Date(last_follow_up),

    age_years = as.numeric(difftime(index_date, birth_date, units = "days")) / 365.25,

    dx_text = str_to_lower(str_trim(paste(dx_main, dx_other))),
    diagnosis_group = case_when(
      str_detect(dx_text, "leuk") ~ "Leukemia",
      str_detect(dx_text, "lymph") ~ "Lymphoma",
      TRUE ~ NA_character_
    ) |> factor(levels = c("Leukemia", "Lymphoma")),

    disease_status = factor(disease_status, levels = c("Remission", "Persistent disease")),
    gvhd_status = factor(gvhd_status, levels = c("Absent", "Present"))
  ) %>%
  mutate(
    os_event = if_else(!is.na(death_date), 1L, 0L),
    os_time_years = as.numeric(difftime(coalesce(death_date, last_follow_up), index_date, units = "days")) / 365.25,

    rfs_event = if_else(!is.na(relapse_date) | !is.na(death_date), 1L, 0L),
    rfs_time_years = as.numeric(difftime(pmin(relapse_date, death_date, last_follow_up, na.rm = TRUE), index_date, units = "days")) / 365.25,

    trm_event = if_else(os_event == 1L & is.na(relapse_date), 1L, 0L),
    trm_time_years = os_time_years
  ) %>%
  filter(!is.na(diagnosis_group), os_time_years >= 0, rfs_time_years >= 0)

Cox Regression & Diagnostics

# ============================================
# COX REGRESSION + DIAGNOSTICS (ENCRYPTED)
# ============================================

library(survival)
library(car)

# Stratify by a non-PH covariate (example)
cox_os <- coxph(
  Surv(os_time_years, os_event) ~ diagnosis_group + age_years + gvhd_status + strata(disease_status),
  data = df
)

summary(cox_os)

# Schoenfeld residual test
ph_test <- cox.zph(cox_os)
print(ph_test)

# Concordance (C-statistic)
summary(cox_os)$concordance

# Multicollinearity screen (approximate)
vif(cox_os)

Plotting

# ============================================
# PLOTTING (ENCRYPTED)
# ============================================

library(survival)
library(survminer)

# Unadjusted KM curves
km_os <- survfit(Surv(os_time_years, os_event) ~ disease_status, data = df)
ggsurvplot(
  km_os,
  data = df,
  conf.int = TRUE,
  pval = TRUE,
  risk.table = TRUE,
  xlab = "Years since index date",
  ylab = "Overall survival probability",
  title = "Overall survival by disease status"
)

km_rfs <- survfit(Surv(rfs_time_years, rfs_event) ~ disease_status, data = df)
ggsurvplot(
  km_rfs,
  data = df,
  conf.int = TRUE,
  pval = TRUE,
  risk.table = TRUE,
  xlab = "Years since index date",
  ylab = "Relapse-free survival probability",
  title = "Relapse-free survival by disease status"
)

# Adjusted survival curves from Cox model
new_df <- data.frame(
  diagnosis_group = levels(df$diagnosis_group),
  age_years = mean(df$age_years, na.rm = TRUE),
  gvhd_status = "Absent",
  disease_status = levels(df$disease_status)[1]
)

fit_adj <- survfit(cox_os, newdata = new_df)
ggsurvplot(
  fit_adj,
  conf.int = FALSE,
  risk.table = FALSE,
  xlab = "Years since index date",
  ylab = "Adjusted survival probability",
  title = "Adjusted survival by diagnosis group"
)
