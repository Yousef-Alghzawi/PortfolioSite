Meta-Analysis • JACC • 2024

LVOTO Prevention Techniques in Transcatheter Mitral Valve Replacement

Published in JACC

Problem

LVOTO is a critical complication in TMVR. Multiple prevention techniques exist but comparative effectiveness data was fragmented across small studies.

Outcome

Presented at TCT 2025. Published in JACC. Reviewers noted rigorous handling of heterogeneous data sources.

Complete Analysis Code

1

Data Cleaning

# ============================================
# DATA CLEANING
# ============================================

# Load required packages
library(readxl)
library(dplyr)
library(tidyr)

# Import raw data
raw_data <- read_excel("data/extracted_studies.xlsx")

# --- Variable-by-variable cleaning ---

# Clean study_id
raw_data$study_id <- trimws(raw_data$study_id)
raw_data$study_id <- toupper(raw_data$study_id)

# Clean technique_type
raw_data$technique_type <- trimws(raw_data$technique_type)
raw_data$technique_type <- tolower(raw_data$technique_type)
raw_data$technique_type[raw_data$technique_type == "lampoon"] <- "laceration"
raw_data$technique_type[raw_data$technique_type == "asa"] <- "ablation"
raw_data$technique_type[raw_data$technique_type == "alcohol septal ablation"] <- "ablation"

# Clean n_intervention (sample size intervention group)
raw_data$n_int <- as.numeric(raw_data$n_int)
raw_data$n_int[raw_data$n_int < 0] <- NA
raw_data$n_int[raw_data$n_int == 0] <- NA

# Clean events_intervention
raw_data$events_int <- as.numeric(raw_data$events_int)
raw_data$events_int[raw_data$events_int < 0] <- NA

# Clean n_control (sample size control group)
raw_data$n_ctrl <- as.numeric(raw_data$n_ctrl)
raw_data$n_ctrl[raw_data$n_ctrl < 0] <- NA
raw_data$n_ctrl[raw_data$n_ctrl == 0] <- NA

# Clean events_control
raw_data$events_ctrl <- as.numeric(raw_data$events_ctrl)
raw_data$events_ctrl[raw_data$events_ctrl < 0] <- NA

# Clean outcome_type
raw_data$outcome_type <- trimws(raw_data$outcome_type)
raw_data$outcome_type <- tolower(raw_data$outcome_type)

# Clean follow_up_months
raw_data$followup_mo <- as.numeric(raw_data$followup_mo)
raw_data$followup_mo[raw_data$followup_mo < 0] <- NA

# Clean publication_year
raw_data$pub_year <- as.numeric(raw_data$pub_year)
raw_data$pub_year[raw_data$pub_year < 2000] <- NA
raw_data$pub_year[raw_data$pub_year > 2024] <- NA

# Clean quality_score (Newcastle-Ottawa)
raw_data$nos_score <- as.numeric(raw_data$nos_score)
raw_data$nos_score[raw_data$nos_score < 0] <- NA
raw_data$nos_score[raw_data$nos_score > 9] <- NA

# --- Create analysis-ready dataset ---

# Filter for primary outcome (LVOTO occurrence)
data_primary <- raw_data[raw_data$outcome_type == "lvoto", ]

# Filter for secondary outcome (hemodynamic success)
data_secondary <- raw_data[raw_data$outcome_type == "hemodynamic", ]

# Remove rows with missing essential variables
data_primary <- data_primary[!is.na(data_primary$n_int), ]
data_primary <- data_primary[!is.na(data_primary$events_int), ]

data_secondary <- data_secondary[!is.na(data_secondary$n_int), ]
data_secondary <- data_secondary[!is.na(data_secondary$events_int), ]

# Verify no events exceed sample size
data_primary$events_int[data_primary$events_int > data_primary$n_int] <- NA
data_primary$events_ctrl[data_primary$events_ctrl > data_primary$n_ctrl] <- NA

data_secondary$events_int[data_secondary$events_int > data_secondary$n_int] <- NA
data_secondary$events_ctrl[data_secondary$events_ctrl > data_secondary$n_ctrl] <- NA

# Remove rows with newly created NAs
data_primary <- na.omit(data_primary)
data_secondary <- na.omit(data_secondary)

# Create subgroup indicators
data_primary$subgroup_technique <- factor(data_primary$technique_type)
data_primary$subgroup_quality <- ifelse(data_primary$nos_score >= 7, "high", "low")
data_primary$subgroup_quality <- factor(data_primary$subgroup_quality)

data_secondary$subgroup_technique <- factor(data_secondary$technique_type)
data_secondary$subgroup_quality <- ifelse(data_secondary$nos_score >= 7, "high", "low")
data_secondary$subgroup_quality <- factor(data_secondary$subgroup_quality)

# Save cleaned datasets
write.csv(data_primary, "data/clean_primary.csv", row.names = FALSE)
write.csv(data_secondary, "data/clean_secondary.csv", row.names = FALSE)

Analysis

# ============================================
# ANALYSIS
# ============================================

# Load required packages
library(meta)
library(metafor)

# Load cleaned data
data_primary <- read.csv("data/clean_primary.csv")
data_secondary <- read.csv("data/clean_secondary.csv")

# --- PRIMARY OUTCOME: LVOTO Occurrence ---

# Main meta-analysis (Peto method for rare events)
meta_primary <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary,
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE,
  method.tau = "DL",
  hakn = TRUE,
  prediction = TRUE
)

# Print primary results
print(summary(meta_primary))

# Heterogeneity statistics for primary outcome
cat("I-squared:", meta_primary$I2, "\n")
cat("Tau-squared:", meta_primary$tau2, "\n")
cat("Q statistic:", meta_primary$Q, "\n")
cat("Q p-value:", meta_primary$pval.Q, "\n")

# --- SECONDARY OUTCOME: Hemodynamic Success ---

# Main meta-analysis
meta_secondary <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_secondary,
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE,
  method.tau = "DL",
  hakn = TRUE,
  prediction = TRUE
)

# Print secondary results
print(summary(meta_secondary))

# Heterogeneity statistics for secondary outcome
cat("I-squared:", meta_secondary$I2, "\n")
cat("Tau-squared:", meta_secondary$tau2, "\n")
cat("Q statistic:", meta_secondary$Q, "\n")
cat("Q p-value:", meta_secondary$pval.Q, "\n")

# --- SUBGROUP ANALYSIS BY TECHNIQUE ---

# Subgroup: Laceration technique
meta_subgroup_laceration <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary[data_primary$technique_type == "laceration", ],
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE
)

# Subgroup: Ablation technique
meta_subgroup_ablation <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary[data_primary$technique_type == "ablation", ],
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE
)

# Subgroup: Resection technique
meta_subgroup_resection <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary[data_primary$technique_type == "resection", ],
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE
)

# Print subgroup results
cat("\n--- SUBGROUP: Laceration ---\n")
print(summary(meta_subgroup_laceration))

cat("\n--- SUBGROUP: Ablation ---\n")
print(summary(meta_subgroup_ablation))

cat("\n--- SUBGROUP: Resection ---\n")
print(summary(meta_subgroup_resection))

# --- SUBGROUP ANALYSIS BY STUDY QUALITY ---

# Subgroup: High quality (NOS >= 7)
meta_subgroup_highqual <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary[data_primary$subgroup_quality == "high", ],
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE
)

# Subgroup: Low quality (NOS < 7)
meta_subgroup_lowqual <- metabin(
  event.e = events_int,
  n.e = n_int,
  event.c = events_ctrl,
  n.c = n_ctrl,
  studlab = study_id,
  data = data_primary[data_primary$subgroup_quality == "low", ],
  sm = "OR",
  method = "Peto",
  fixed = FALSE,
  random = TRUE
)

# Print quality subgroup results
cat("\n--- SUBGROUP: High Quality ---\n")
print(summary(meta_subgroup_highqual))

cat("\n--- SUBGROUP: Low Quality ---\n")
print(summary(meta_subgroup_lowqual))

# --- SENSITIVITY ANALYSES ---

# Leave-one-out analysis
loo_primary <- metainf(meta_primary, pooled = "random")
print(loo_primary)

# Influence diagnostics
influence_primary <- influence(meta_primary)
print(influence_primary)

# --- PUBLICATION BIAS ---

# Egger's test for primary outcome
egger_primary <- metabias(meta_primary, method.bias = "Egger")
cat("\n--- EGGER'S TEST ---\n")
print(egger_primary)

# Peters' test (preferred for binary outcomes)
peters_primary <- metabias(meta_primary, method.bias = "Peters")
cat("\n--- PETERS' TEST ---\n")
print(peters_primary)

# Trim and fill analysis
trimfill_primary <- trimfill(meta_primary)
cat("\n--- TRIM AND FILL ---\n")
print(summary(trimfill_primary))
