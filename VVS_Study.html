GLM & Proportion Testing

Application of Generalized Linear Models (GLM) and proportion tests to compare Siteal event rates.

DATA CLEANING & PREPARING

library(readxl)
library(tidyverse)
library(janitor)

# Load data (Encrypted path)
data_path <- "encrypted_path/Event A_data.xlsx"
df <- read_excel(data_path) %>% clean_names()

# Data Pre-processing
# Converting categorical variables to factors
df <- df %>%
  mutate(
    Event A_history = factor(Event A_history, levels = c("No", "Yes")),
    trigger_type = factor(trigger_type),
    gender = factor(gender),
    age_group = factor(age_group, ordered = TRUE)
  )

INFERENTIALS 

# Generalized Linear Models (GLM)
# Predicting Event A Recurrence
glm_model <- glm(recurrence ~ gender + age_group + trigger_type,
                 family = binomial(link = "logit"),
                 data = df)

summary(glm_model)
