Mortality Trends in CNS Inflammatory Diseases

Published in Clinical Neurology and Neurosurgery

Problem

20-year population-level mortality analysis for encephalitis, myelitis, and encephalomyelitis requiring joinpoint regression and demographic stratification.

Accepted without statistical revision. Published in Clinical Neurology and Neurosurgery.

Data Cleaning

# ============================================
# DATA CLEANING
# ============================================

# Load required packages
library(readr)
library(dplyr)

# Import CDC WONDER mortality data
raw_data <- read_delim("data/cdc_wonder_extract.txt", delim = "\t")

# --- Variable-by-variable cleaning ---

# Clean year
raw_data$year <- as.numeric(raw_data$year)
raw_data$year[raw_data$year < 1999] <- NA
raw_data$year[raw_data$year > 2020] <- NA

# Clean deaths count
raw_data$deaths <- as.numeric(raw_data$deaths)
raw_data$deaths[raw_data$deaths < 0] <- NA
raw_data$deaths[is.na(raw_data$deaths)] <- 0

# Clean population
raw_data$population <- as.numeric(raw_data$population)
raw_data$population[raw_data$population < 0] <- NA

# Clean age_group
raw_data$age_grp <- trimws(raw_data$age_grp)
raw_data$age_grp <- gsub(" years", "", raw_data$age_grp)
raw_data$age_grp <- gsub("-", "_", raw_data$age_grp)

# Standardize age groups
raw_data$age_grp[raw_data$age_grp == "0_4"] <- "00_04"
raw_data$age_grp[raw_data$age_grp == "5_9"] <- "05_09"
raw_data$age_grp[raw_data$age_grp == "85+"] <- "85_plus"

raw_data$age_grp <- factor(raw_data$age_grp, levels = c(
  "00_04", "05_09", "10_14", "15_19", "20_24", "25_29", "30_34", 
  "35_39", "40_44", "45_49", "50_54", "55_59", "60_64", "65_69", 
  "70_74", "75_79", "80_84", "85_plus"
))

# Clean sex
raw_data$sex <- trimws(raw_data$sex)
raw_data$sex <- tolower(raw_data$sex)
raw_data$sex[raw_data$sex == "m"] <- "male"
raw_data$sex[raw_data$sex == "f"] <- "female"
raw_data$sex <- factor(raw_data$sex, levels = c("male", "female"))

# Clean race_ethnicity
raw_data$race_eth <- trimws(raw_data$race_eth)
raw_data$race_eth <- tolower(raw_data$race_eth)
raw_data$race_eth[raw_data$race_eth == "white"] <- "white_nh"
raw_data$race_eth[raw_data$race_eth == "black"] <- "black_nh"
raw_data$race_eth[raw_data$race_eth == "hispanic"] <- "hispanic"
raw_data$race_eth[raw_data$race_eth == "asian/pacific islander"] <- "asian_pi"
raw_data$race_eth[raw_data$race_eth == "american indian/alaska native"] <- "aian"
raw_data$race_eth <- factor(raw_data$race_eth)

# Clean ICD code category
raw_data$icd_cat <- trimws(raw_data$icd_cat)
raw_data$icd_cat <- tolower(raw_data$icd_cat)
raw_data$icd_cat[raw_data$icd_cat == "g04"] <- "encephalitis"
raw_data$icd_cat[raw_data$icd_cat == "g05"] <- "encephalitis_secondary"
raw_data$icd_cat[raw_data$icd_cat == "g37"] <- "demyelinating"
raw_data$icd_cat <- factor(raw_data$icd_cat)

# Calculate crude mortality rate (per 100,000)
raw_data$crude_rate <- (raw_data$deaths / raw_data$population) * 100000

# Remove rows with missing population
raw_data <- raw_data[!is.na(raw_data$population), ]
raw_data <- raw_data[raw_data$population > 0, ]

# --- Create aggregate datasets ---

# Aggregate by year (overall)
data_year <- aggregate(
  cbind(deaths, population) ~ year, 
  data = raw_data, 
  FUN = sum
)
data_year$crude_rate <- (data_year$deaths / data_year$population) * 100000

# Aggregate by year and sex
data_year_sex <- aggregate(
  cbind(deaths, population) ~ year + sex, 
  data = raw_data, 
  FUN = sum
)
data_year_sex$crude_rate <- (data_year_sex$deaths / data_year_sex$population) * 100000

# Aggregate by year and race/ethnicity
data_year_race <- aggregate(
  cbind(deaths, population) ~ year + race_eth, 
  data = raw_data, 
  FUN = sum
)
data_year_race$crude_rate <- (data_year_race$deaths / data_year_race$population) * 100000

# Aggregate by year and age group
data_year_age <- aggregate(
  cbind(deaths, population) ~ year + age_grp, 
  data = raw_data, 
  FUN = sum
)
data_year_age$crude_rate <- (data_year_age$deaths / data_year_age$population) * 100000

# --- Age-adjusted rates ---
# Using 2000 US Standard Population

std_pop_2000 <- data.frame(
  age_grp = c("00_04", "05_09", "10_14", "15_19", "20_24", "25_29", "30_34", 
              "35_39", "40_44", "45_49", "50_54", "55_59", "60_64", "65_69", 
              "70_74", "75_79", "80_84", "85_plus"),
  std_weight = c(0.069135, 0.072533, 0.073032, 0.072169, 0.066478, 0.064530,
                 0.071044, 0.080762, 0.081851, 0.072118, 0.062716, 0.048454,
                 0.038793, 0.034264, 0.031773, 0.026999, 0.017842, 0.015508)
)

# Merge with age-specific data
data_merged <- merge(data_year_age, std_pop_2000, by = "age_grp")

# Calculate age-specific weighted rates
data_merged$weighted_rate <- data_merged$crude_rate * data_merged$std_weight

# Aggregate to get age-adjusted rate by year
data_aar <- aggregate(weighted_rate ~ year, data = data_merged, FUN = sum)
names(data_aar)[2] <- "aar"

# Save cleaned datasets
write.csv(data_year, "data/clean_yearly.csv", row.names = FALSE)
write.csv(data_year_sex, "data/clean_yearly_sex.csv", row.names = FALSE)
write.csv(data_year_race, "data/clean_yearly_race.csv", row.names = FALSE)
write.csv(data_aar, "data/clean_aar.csv", row.names = FALSE)

Analysis

# ============================================
# ANALYSIS
# ============================================

# Load required packages
library(joinpoint)  # NCI Joinpoint package or equivalent
library(segmented)

# Load cleaned data
data_year <- read.csv("data/clean_yearly.csv")
data_year_sex <- read.csv("data/clean_yearly_sex.csv")
data_year_race <- read.csv("data/clean_yearly_race.csv")
data_aar <- read.csv("data/clean_aar.csv")

# --- OVERALL TREND ANALYSIS ---

# Linear regression: Overall crude rate
lm_overall <- lm(crude_rate ~ year, data = data_year)
print(summary(lm_overall))

# Calculate Annual Percent Change (APC)
# APC = (exp(beta) - 1) * 100
beta_overall <- coef(lm_overall)["year"]
apc_overall <- (exp(beta_overall) - 1) * 100
cat("Overall APC:", round(apc_overall, 2), "%\n")

# Joinpoint regression: Overall
# Using segmented package for changepoint detection
seg_overall <- segmented(lm_overall, seg.Z = ~year, npsi = 1)
print(summary(seg_overall))

# Extract breakpoints
breakpoints_overall <- seg_overall$psi[, "Est."]
cat("Joinpoint(s):", breakpoints_overall, "\n")

# Calculate APC for each segment
slope_seg1 <- slope(seg_overall)$year[1, "Est."]
slope_seg2 <- slope(seg_overall)$year[2, "Est."]

apc_seg1 <- (exp(slope_seg1) - 1) * 100
apc_seg2 <- (exp(slope_seg2) - 1) * 100

cat("APC Segment 1:", round(apc_seg1, 2), "%\n")
cat("APC Segment 2:", round(apc_seg2, 2), "%\n")

# --- TREND BY SEX ---

# Male subset
data_male <- data_year_sex[data_year_sex$sex == "male", ]
lm_male <- lm(crude_rate ~ year, data = data_male)
print(summary(lm_male))

beta_male <- coef(lm_male)["year"]
apc_male <- (exp(beta_male) - 1) * 100
cat("Male APC:", round(apc_male, 2), "%\n")

# Female subset
data_female <- data_year_sex[data_year_sex$sex == "female", ]
lm_female <- lm(crude_rate ~ year, data = data_female)
print(summary(lm_female))

beta_female <- coef(lm_female)["year"]
apc_female <- (exp(beta_female) - 1) * 100
cat("Female APC:", round(apc_female, 2), "%\n")

# Joinpoint: Male
seg_male <- segmented(lm_male, seg.Z = ~year, npsi = 1)
print(summary(seg_male))

# Joinpoint: Female
seg_female <- segmented(lm_female, seg.Z = ~year, npsi = 1)
print(summary(seg_female))

# --- TREND BY RACE/ETHNICITY ---

# White non-Hispanic
data_white <- data_year_race[data_year_race$race_eth == "white_nh", ]
lm_white <- lm(crude_rate ~ year, data = data_white)
beta_white <- coef(lm_white)["year"]
apc_white <- (exp(beta_white) - 1) * 100
cat("White NH APC:", round(apc_white, 2), "%\n")

# Black non-Hispanic
data_black <- data_year_race[data_year_race$race_eth == "black_nh", ]
lm_black <- lm(crude_rate ~ year, data = data_black)
beta_black <- coef(lm_black)["year"]
apc_black <- (exp(beta_black) - 1) * 100
cat("Black NH APC:", round(apc_black, 2), "%\n")

# Hispanic
data_hispanic <- data_year_race[data_year_race$race_eth == "hispanic", ]
lm_hispanic <- lm(crude_rate ~ year, data = data_hispanic)
beta_hispanic <- coef(lm_hispanic)["year"]
apc_hispanic <- (exp(beta_hispanic) - 1) * 100
cat("Hispanic APC:", round(apc_hispanic, 2), "%\n")

# Asian/Pacific Islander
data_asian <- data_year_race[data_year_race$race_eth == "asian_pi", ]
lm_asian <- lm(crude_rate ~ year, data = data_asian)
beta_asian <- coef(lm_asian)["year"]
apc_asian <- (exp(beta_asian) - 1) * 100
cat("Asian/PI APC:", round(apc_asian, 2), "%\n")

# --- AGE-ADJUSTED RATE TREND ---

lm_aar <- lm(aar ~ year, data = data_aar)
print(summary(lm_aar))

beta_aar <- coef(lm_aar)["year"]
apc_aar <- (exp(beta_aar) - 1) * 100
cat("Age-Adjusted APC:", round(apc_aar, 2), "%\n")

# Joinpoint: Age-adjusted
seg_aar <- segmented(lm_aar, seg.Z = ~year, npsi = 1)
print(summary(seg_aar))

# --- COMPARISON TESTS ---

# Compare male vs female trends (interaction test)
data_year_sex$sex <- factor(data_year_sex$sex)
lm_sex_interaction <- lm(crude_rate ~ year * sex, data = data_year_sex)
print(summary(lm_sex_interaction))

# Test for significant difference in slopes
anova_sex <- anova(lm_sex_interaction)
print(anova_sex)

# --- RATE RATIOS ---

# Calculate mean rates by group for comparison
mean_rate_male <- mean(data_male$crude_rate)
mean_rate_female <- mean(data_female$crude_rate)
rr_sex <- mean_rate_male / mean_rate_female
cat("Rate Ratio (M/F):", round(rr_sex, 2), "\n")

mean_rate_white <- mean(data_white$crude_rate)
mean_rate_black <- mean(data_black$crude_rate)
rr_race <- mean_rate_black / mean_rate_white
cat("Rate Ratio (Black/White):", round(rr_race, 2), "\n")

# --- AVERAGE ANNUAL PERCENT CHANGE (AAPC) ---

# For studies with joinpoints, calculate AAPC as weighted average
# Weight = years in each segment

years_seg1 <- breakpoints_overall - min(data_year$year)
years_seg2 <- max(data_year$year) - breakpoints_overall
total_years <- max(data_year$year) - min(data_year$year)

aapc_overall <- (apc_seg1 * years_seg1 + apc_seg2 * years_seg2) / total_years
cat("AAPC (Overall):", round(aapc_overall, 2), "%\n")

png("figures/trend_overall.png", width = 10, height = 6, units = "in", res = 300)

# Extract fitted values from segmented model
data_year$fitted <- predict(seg_overall)

ggplot(data_year, aes(x = year, y = crude_rate)) +
  geom_point(size = 3, color = "#10B981") +
  geom_line(aes(y = fitted), color = "#F97316", linewidth = 1.5) +
  geom_vline(xintercept = breakpoints_overall, linetype = "dashed", color = "#8B5CF6", linewidth = 1) +
  scale_x_continuous(breaks = seq(1999, 2020, by = 2)) +
  scale_y_continuous(labels = number_format(accuracy = 0.01)) +

    title = "Mortality Rate Trends: CNS Inflammatory Diseases",

    plot.title = element_text(size = 14, face = "bold"),
